\textbf{Experimental fact} & \textbf{Our verdict} & \textbf{Why it follows}\\
\midrule
\textbf{Anhydrous (5.5\,\AA) is not SC}\srcref{Foo 2003} & \vok &
  $\alpha>0$: single stiff well ($\hbar\omega\!\approx\!30$\,meV),
  $N(0)\!\approx\!0$ --- no 2DEG, nothing to pair.\\
\textbf{Monolayer hydrate (6.9\,\AA) is not SC}\srcref{Foo 2003} & \vok &
  Past the window: $\lambda\!\approx\!20\!\gg\!2$, the alkali self-traps
  (polaronic), so SC is killed.\\
\textbf{Bilayer hydrate (9.9\,\AA) is SC, $T_c\!=\!4.5$\,K}\srcref{Takada 2003}
  & \vpred &
  Bare geometry is polaronic here; SC needs water to soften the Na well
  back into the window --- untested.\\
\textbf{No superconducting Li analogue} & \vok &
  $c^{\ast}_{\mathrm{Li}}\!\le\!5.5$\,\AA\ and lighter mass $\Rightarrow$
  stiffer mode; Li never enters the soft-mode window.\\
\textbf{Magnetic phase borders the dome}\srcref{expt.} & \vok &
  Local moment turns on exactly at the $\alpha<0$ well transition (Fig.~4),
  coincident with the dome.\\
\textbf{Superconducting $T_c$ is only a few K}\srcref{Takada 2003} & \vok &
  Allen--Dynes with the DFT $\omega_{\mathrm{eff}},\lambda$ gives peak
  $T_c\!\approx\!14$\,K (band 11--23) --- the right scale.\\

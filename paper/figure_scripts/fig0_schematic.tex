% Figure 1 - Mechanism schematic (paper visual anchor), single-column (86 mm).
% Na_xCoO2.yH2O: how the CoO2-CoO2 spacing (hydration) drives the alkali
% gallery mode from a stiff single well (no 2DEG) through a soft double well
% (2DEG on, SC window) to a deep polaronic double well at the bilayer spacing
% where water re-softens the well back into the pairing window.
%
% Design note: this figure is placed at \columnwidth (86 mm). It is therefore
% drawn at a small natural width (~9 cm) with LARGE fonts (>= \footnotesize)
% and minimal text, so every label is legible at print size.  Build:
%   pdflatex fig0_schematic.tex
\documentclass[10pt,border=2pt]{standalone}
\usepackage{tikz}
\usepackage{amsmath}
\usepackage{mathptmx}          % Times-like serif to match the APS body text
\usetikzlibrary{arrows.meta,positioning,decorations.pathreplacing,
                decorations.pathmorphing,backgrounds,fit,calc}

% ---- palette (schematic; physics-encoding only) -----------------------------
\definecolor{slab}{HTML}{4A5A63}     % CoO2 slab (neutral structural slate)
\definecolor{slabtxt}{HTML}{FFFFFF}
\definecolor{na}{HTML}{E58A00}       % Na ion (mobile alkali)
\definecolor{nadk}{HTML}{9A5B00}
\definecolor{egas}{HTML}{2A78D6}     % 2DEG carriers (blue)
\definecolor{ink}{HTML}{0B0B0B}
\definecolor{sec}{HTML}{52514E}
\definecolor{good}{HTML}{0C8A2E}
\definecolor{crit}{HTML}{C23A38}
\definecolor{amber}{HTML}{B07600}
\definecolor{winfill}{HTML}{F8E2B4}  % SC-window tint (FILL_AMBER)
\definecolor{gasfill}{HTML}{DCEBFC}  % gas-on tint (light FILL_BLUE)

\begin{document}
\begin{tikzpicture}[x=1cm,y=1cm,line join=round,
    slabstyle/.style={draw=slab!80!black,line width=0.5pt,fill=slab},
    naion/.style={circle,draw=nadk,line width=0.4pt,fill=na,
        minimum size=4.6pt,inner sep=0pt},
    elec/.style={circle,fill=egas,inner sep=0pt,minimum size=1.5pt},
  ]

  % ---- geometry -------------------------------------------------------------
  \def\P{3.05}         % column pitch
  \def\W{1.24}         % slab half-width
  \def\hs{0.34}        % slab thickness
  \def\ww{1.05}        % well half-width
  \def\wh{1.05}        % well height
  \pgfmathsetmacro\Xmax{2*\P}

  % crystal vertical anchors
  \def\ycen{0}         % gallery mid-plane
  \def\ywell{-2.35}    % well baseline
  \def\yhead{1.62}     % header baseline
  \def\yarrow{2.55}    % top arrow

  % =============================================================== CRYSTALS
  % #1 x-origin  #2 half-gap  #3 Na-disp  #4 n_elec  #5 gasfill?
  \foreach \ox/\hg/\disp/\ne/\gas in {0/0.40/0/0/0, \P/0.62/0.34/5/1, {2*\P}/0.86/0.54/6/1}{
    \begin{scope}[shift={(\ox,\ycen)}]
      % gallery tint where the 2DEG lives
      \ifnum\gas>0
        \fill[gasfill] (-\W,-\hg) rectangle (\W,\hg);
      \fi
      % CoO2 slabs
      \fill[slabstyle] (-\W,-\hg-\hs) rectangle (\W,-\hg);
      \fill[slabstyle] (-\W,\hg) rectangle (\W,\hg+\hs);
      \node[slabtxt,font=\footnotesize] at (0,-\hg-\hs/2) {CoO$_2$};
      \node[slabtxt,font=\footnotesize] at (0,\hg+\hs/2) {CoO$_2$};
      % 2DEG electron sheets on gallery-facing surfaces
      \ifnum\ne>0
        \foreach \k in {1,...,\ne}{
          \pgfmathsetmacro\ex{-\W+0.22+(\k-1)*(2*\W-0.44)/(\ne-1)}
          \node[elec] at (\ex,-\hg+0.09){};
          \node[elec] at (\ex,\hg-0.09){};
        }
      \fi
      % Na ion (+ bistable ghost + double-well hop arrow)
      \ifdim\disp pt>0pt
        \node[naion,fill=na!35,draw=na!45!white] at (-\disp,0){};
        \draw[<->,sec,line width=0.35pt] (-\disp+0.14,0) -- (\disp-0.14,0);
      \fi
      \node[naion] at (\disp,0){};
      \node[nadk,font=\footnotesize,anchor=south,inner sep=1pt] at (\disp,0.12){Na$^+$};
    \end{scope}
  }
  % single 2DEG label (columns 2-3 share the carrier physics)
  \node[egas,font=\footnotesize] at (1.5*\P,-1.52) {2DEG $e^-$ (carriers)};

  % =============================================================== WELLS
  % V(delta): single / shallow-double / deep-double, with quantized levels
  \foreach \ox/\type in {0/single, \P/soft, {2*\P}/deep}{
    \begin{scope}[shift={(\ox,\ywell)}]
      \draw[sec,line width=0.4pt,-{Latex[length=2.6pt]}]
        (-\ww-0.12,-\wh) -- (\ww+0.16,-\wh);
      \draw[sec,line width=0.4pt,-{Latex[length=2.6pt]}]
        (0,-\wh) -- (0,0.18);
      \node[sec,font=\footnotesize,anchor=north] at (\ww+0.14,-\wh+0.02){$\delta$};
      \node[sec,font=\footnotesize,anchor=east] at (-0.05,0.16){$V$};

      \ifnum\pdfstrcmp{\type}{single}=0
        \draw[ink,line width=0.9pt,smooth,domain=-0.92:0.92,samples=50]
          plot (\x,{-\wh+1.15*\x*\x});
        \pgfmathsetmacro\ea{-\wh+0.26}
        \draw[egas,line width=0.9pt] (-0.38,\ea)--(0.38,\ea);
        \node[egas,font=\footnotesize,anchor=south,inner sep=0.6pt] at (0,\ea){$E_0$};
      \fi
      \ifnum\pdfstrcmp{\type}{soft}=0
        \draw[ink,line width=0.9pt,smooth,domain=-1.02:1.02,samples=80]
          plot (\x,{-\wh+0.60+(-1.60*\x*\x+1.10*\x*\x*\x*\x)});
        \pgfmathsetmacro\eb{-\wh+0.16}
        \draw[egas,line width=0.9pt] (-0.80,\eb)--(0.80,\eb);
        \node[egas,font=\footnotesize,anchor=south,inner sep=0.6pt] at (0,\eb){$E_0{,}E_1$};
      \fi
      \ifnum\pdfstrcmp{\type}{deep}=0
        \draw[ink,line width=0.9pt,smooth,domain=-1.02:1.02,samples=80]
          plot (\x,{-\wh+0.85+(-2.7*\x*\x+1.85*\x*\x*\x*\x)});
        \pgfmathsetmacro\ec{-\wh+0.22}
        \draw[egas,line width=0.9pt] (-0.78,\ec)--(0.78,\ec);
        \node[egas,font=\footnotesize,anchor=south,inner sep=0.6pt] at (0,\ec){$E_0{,}E_1$};
      \fi
    \end{scope}
  }

  % =============================================================== HEADERS
  % #1 x  #2 spacing(bold)  #3 descriptor  #4 verdict  #5 verdict-colour
  \foreach \ox/\sp/\desc/\vd/\vc in {
      0/{$c=5.5$\,\AA}/{anhydrous}/{not SC}/crit,
      \P/{$c\approx6.4$\,\AA}/{SC window}/{SC window}/good,
      {2*\P}/{$c=9.9$\,\AA}/{bilayer hydrate}/{SC 4.5\,K}/amber}{
    \node[ink,font=\footnotesize\bfseries,anchor=south] at (\ox,\yhead){\sp};
    \node[sec,font=\footnotesize,anchor=south] at (\ox,\yhead-0.34){\desc};
    \node[\vc,font=\footnotesize\bfseries,anchor=north,draw=\vc,
          rounded corners=1.5pt,inner sep=2.2pt,line width=0.5pt]
      at (\ox,\ywell-\wh-0.18){\vd};
  }
  % water re-softening note under the bilayer column
  \node[sec,font=\footnotesize,anchor=north,text width=2.9cm,align=center]
    at (2*\P,\ywell-\wh-0.68){water re-softens\\the well};

  % =============================================================== TOP ARROW
  \draw[-{Latex[length=5pt,width=5pt]},slab,line width=1.3pt]
    (-\W-0.25,\yarrow) -- (\Xmax+\W+0.25,\yarrow);
  \node[ink,font=\footnotesize\bfseries,anchor=south] at (\P,\yarrow+0.06)
    {increasing CoO$_2$--CoO$_2$ spacing $c$ \ (hydration)};

\end{tikzpicture}
\end{document}
